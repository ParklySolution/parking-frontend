        console.log('🔍 Utente loggato:', user?.email);

          console.error('❌ Nessun utente loggato');

          return;

          .select('tenant_id')
          .eq('id', user.id)

          console.error('❌ Errore recupero profilo:', error);
          setLoadingTenant(false);
          return;

          console.error('❌ Nessun tenant_id associato al profilo');
        }

        console.error('❌ Errore recupero tenant ID:', err);

        setLoadingTenant(false);
      }

        setLogs(data);

        if (data.length > 0) {
          const last = data[0];
          setGateStatus(last.allowed ? "allowed" : "blocked");
        } else {
          setGateStatus("idle");
        }

        console.error("Errore caricamento log", e);

        setLoading(false);

type GateStatus = "idle" | "checking" | "allowed" | "blocked";

  const navigate = useNavigate();

  // Stati per tenant e autenticazione
  const [tenantId, setTenantId] = useState<string | null>(null);
  const [loadingTenant, setLoadingTenant] = useState(true);

  const [logs, setLogs] = useState<PlateLog[]>([]);
  const [loading, setLoading] = useState(true);
  const [gateStatus, setGateStatus] = useState<GateStatus>("idle");

  const [selectedPlate, setSelectedPlate] = useState<string | null>(null);
  const [showExitModal, setShowExitModal] = useState(false);

  /* ===============================
     RECUPERA TENANT ID DALLA SESSIONE
     =============================== */
  useEffect(() => {
    const getTenantId = async () => {
      try {
        const { data: { user } } = await supabase.auth.getUser();
        console.log('🔍 Utente loggato:', user?.email);
        
        if (!user) {
          console.error('❌ Nessun utente loggato');
          setLoadingTenant(false);
          return;
        }

        // 🔥 CORREZIONE: Prendi tenant_id direttamente dai metadata dell'utente
        const tenantIdFromMetadata = user.user_metadata?.tenant_id;
        
        if (tenantIdFromMetadata) {
          console.log('🏢 Tenant ID da metadata:', tenantIdFromMetadata);
          setTenantId(tenantIdFromMetadata);
          setLoadingTenant(false);
          return;
        }

        // Fallback: cerca nella tabella admin_profiles
        const { data: profile, error } = await supabase
          .from('admin_profiles')
          .select('company_id')
          .eq('auth_user_id', user.id)
          .single();

          console.error('❌ Errore recupero profilo:', error);
          setLoadingTenant(false);
          return;
        }

        if (profile?.company_id) {
          console.log('🏢 Tenant ID da admin_profiles:', profile.company_id);
          setTenantId(profile.company_id);

          console.error('❌ Nessun tenant_id trovato per l\'utente');
        }
      } catch (err) {
        console.error('❌ Errore recupero tenant ID:', err);
      } finally {
        setLoadingTenant(false);
      }
    };

    getTenantId();
  }, []);

  /* ===============================
     CARICAMENTO LOG TARGHE REALI
     =============================== */
  useEffect(() => {
    const loadRealLogs = async () => {
      if (!tenantId) return;
      
      setLoading(true);
      try {
        setGateStatus("checking");

        // 🔥 CHIAMATA REALE A SUPABASE
        const { data: sessions, error } = await supabase
          .from('parking_sessions')
          .select(`
            id,
            plate_number,
            entry_time,
            exit_time,
            status,
            tenant_id
          `)
          .eq('tenant_id', tenantId)
          .eq('status', 'active')
          .order('entry_time', { ascending: false })
          .limit(10);

        if (error) {
          console.error('Errore caricamento sessioni:', error);
          setLogs([]);
          setGateStatus("blocked");
        } else if (sessions && sessions.length > 0) {
          // Converti in formato PlateLog
          const formattedLogs: PlateLog[] = sessions.map(session => ({
            plate: session.plate_number,
            timestamp: session.entry_time,
            allowed: true,
            sessionId: session.id
          }));
          setLogs(formattedLogs);
          setGateStatus("allowed");
        } else {
          setLogs([]);
          setGateStatus("idle");
        }
      } catch (e) {
        console.error("Errore caricamento log", e);
        setGateStatus("blocked");
      } finally {
        setLoading(false);
      }
    };

    if (tenantId) {
      loadRealLogs();
      const interval = setInterval(loadRealLogs, 10000); // Aggiorna ogni 10 secondi
      return () => clearInterval(interval);
    }
  }, [tenantId]);

  /* ===============================

        <div className="dashboard">

          </div>

                opacity: selectedPlate ? 1 : 0.7
              }}
            >

              {selectedPlate ? `ENTRATA ${selectedPlate}` : "NUOVA ENTRATA"}
            </button>

              <div className="date">
                <FaClock style={{ marginRight: "8px", opacity: 0.7 }} />

                  weekday: "long",
